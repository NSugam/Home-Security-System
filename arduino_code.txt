#define DETECT 2 // pin 2 for  sensor
#define ACTION 8 // pin 8 for action for buzzer

void setup() {
  Serial.begin(9600);
  Serial.println("Home Security System");
  pinMode(DETECT, INPUT);//define DEFECT input pin
  pinMode(ACTION, OUTPUT);//define ACTION output pin
}

void loop() {

  int detected = digitalRead(DETECT);// read Laser sensor
 
  if( detected == LOW)
  {
    digitalWrite(ACTION,HIGH);// set the buzzer ON
    Serial.println("Laser Cut Detected!");

  }
  else {
      Serial.println("Welcome");
      digitalWrite(ACTION,LOW); // Set the buzzer OFF
    Serial.println("No laser cut");
  }
  delay(100);
}
